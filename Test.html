# nbi:hide_in
import numpy as np
import nbinteract as nbi


options = {
    'xlabel': 'x',
    'ylabel': 'y',
    'ylim': (-4, 4),
    'xlim': (-4, 4),
    'animation_duration': 250
}

x = np.linspace(-5, 5, num=30)

def f(x, m, b):
    return m * x + b

fig = nbi.Figure(options=options)

fig.scatter(x,x*x)
fig.line(x, f, m=(-2.0, 2.0, .2), b=(-3, 3, 0.5))
